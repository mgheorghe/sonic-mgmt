def get_routing_type_from_str(routing_type_str):
    # Construct the full enum name dynamically
    enum_name = f"ROUTING_TYPE_{routing_type_str.upper()}"
    try:
        # Use getattr to get the enum member
        return getattr(RoutingType, enum_name)
    except AttributeError:
        raise ValueError(f"'{enum_name}' is not a valid routing type")

    # 4_to_6 uses small cap so cannot use dynamic naming
    if action_type_str == "4_to_6":
        return ActionType.ACTION_TYPE_4_to_6
    else:
        # Construct the full enum name dynamically
        enum_name = f"ACTION_TYPE_{action_type_str.upper()}"
        try:
            # Use getattr to get the enum member
            return getattr(ActionType, enum_name)
        except AttributeError:
            raise ValueError(f"'{enum_name}' is not a valid action type")

def vnet_mapping_from_json(json_obj):
    pb = VnetMapping()
    if json_obj.get("action_type") is not None:
        pb.action_type = get_routing_type_from_str(json_obj["action_type"])
    pb.underlay_ip.ipv4 = socket.htonl(int(ipaddress.IPv4Address(json_obj["underlay_ip"])))
    if json_obj.get("mac_address") is not None:
        pb.mac_address = bytes.fromhex(json_obj["mac_address"].replace(":", ""))
    if json_obj.get("metering_class") is not None:
        pb.metering_class = uint64(json_obj["metering_class"])
    if json_obj.get("override_meter") is not None:
        pb.override_meter = json_obj["override_meter"] == "true"
    if json_obj.get("use_dst_vni") is not None:
        pb.use_dst_vni = json_obj["use_dst_vni"] == "true"
    if json_obj.get("use_pl_sip_eni") is not None:
        pb.use_pl_sip_eni = json_obj["use_pl_sip_eni"] == "true"

    if json_obj.get("routing_appliance_id") is not None:
        pb.routing_appliance_id = int(json_obj["routing_appliance_id"])
    if json_obj.get("metering_class_or") is not None:
        pb.metering_class_or = int(json_obj["metering_class_or"] )

    # these are IPv6 prefixes
    if json_obj.get("overlay_sip_prefix") is not None:
        addr_prefix = json_obj["overlay_sip_prefix"].split('/')
        pb.overlay_sip_prefix.ip.ipv6 = ipaddress.IPv6Address(addr_prefix[0]).packed
        pb.overlay_sip_prefix.mask.ipv6 = ipaddress.IPv6Address(addr_prefix[1]).packed
    if json_obj.get("overlay_dip_prefix") is not None:
        addr_prefix = json_obj["overlay_dip_prefix"].split('/')
        pb.overlay_dip_prefix.ip.ipv6 = ipaddress.IPv6Address(addr_prefix[0]).packed
        pb.overlay_dip_prefix.mask.ipv6 = ipaddress.IPv6Address(addr_prefix[1]).packed
    if json_obj.get("tunnel") is not None:
        pb.tunnel = json_obj["tunnel"]
    if json_obj.get("routing_type") is not None:
        pb.routing_type = get_routing_type_from_str(json_obj["routing_type"])

    return pb

def eni_from_json(json_obj):
    pb = Eni()
    pb.eni_id = json_obj["eni_id"]
    pb.mac_address = bytes.fromhex(json_obj["mac_address"].replace(":", ""))
    pb.underlay_ip.ipv4 = socket.htonl(int(ipaddress.IPv4Address(json_obj["underlay_ip"])))
    pb.admin_state =  2 if json_obj["admin_state"] == "enabled" else 1
    pb.vnet = json_obj["vnet"]
    if json_obj.get("qos") is not None:
        pb.qos = json_obj["qos"]
    #Private Link encoding for IPv6 SIP transpositions
    if json_obj.get("pl_sip_encoding") is not None:
        addr_prefix = json_obj["pl_sip_encoding"].split('/')
        pb.pl_sip_encoding.ip.ipv6 = ipaddress.IPv6Address(addr_prefix[0]).packed
        pb.pl_sip_encoding.mask.ipv6 = ipaddress.IPv6Address(addr_prefix[1]).packed

    #Underlay SIP (ST GW VIP) to be used for all private link transformation for this ENI, IPv4 address
    if json_obj.get("pl_underlay_sip") is not None:
        pb.pl_underlay_sip.ipv4 = socket.htonl(int(ipaddress.IPv4Address(json_obj["pl_underlay_sip"])))
    #IPv4 meter policy ID
    if json_obj.get("v4_meter_policy_id") is not None:
        pb.v4_meter_policy_id = json_obj["v4_meter_policy_id"]
    #IPv6 meter policy ID
    if json_obj.get("v6_meter_policy_id") is not None:
        pb.v6_meter_policy_id = json_obj["v6_meter_policy_id"]
    if json_obj.get("disable_fast_path_icmp_flow_redirection") is not None:
        pb.disable_fast_path_icmp_flow_redirection = json_obj["disable_fast_path_icmp_flow_redirection"] == "true"

    return pb

def route_from_json(json_obj):

    pb = Route()
    if json_obj.get("action_type") is not None:
        pb.action_type = get_routing_type_from_str(json_obj["action_type"])
    if json_obj.get("routing_type") is not None:
        pb.routing_type = get_routing_type_from_str(json_obj["routing_type"])

    if json_obj.get("vnet") is not None:
        if pb.action_type != RoutingType.ROUTING_TYPE_VNET and pb.routing_type != RoutingType.ROUTING_TYPE_VNET:
            raise Exception("Action or routing type needs to be vnet for this attribute")
        pb.vnet = json_obj["vnet"]

    if json_obj.get("vnet_direct") is not None:
        if pb.action_type != RoutingType.ROUTING_TYPE_VNET_DIRECT and pb.routing_type != RoutingType.ROUTING_TYPE_VNET_DIRECT:
            raise Exception("Action or routing type needs to be vnet_direct for this attribute")
        pb_vnet_dir = json_obj["vnet_direct"]
        pb.vnet_direct.vnet = pb_vnet_dir.get("vnet")
        if pb_vnet_dir.get("overlay_ip") is not None:
            if type(IP(pb_vnet_dir["overlay_ip"])) is ipaddress.IPv4Address:
                pb.vnet_direct.overlay_ip.ipv4 = socket.htonl(int(ipaddress.IPv4Address(pb_vnet_dir["overlay_ip"])))
            elif type(IP(pb_vnet_dir["overlay_ip"])) is ipaddress.IPv6Address:
                pb.vnet_direct.overlay_ip.ipv6 = ipaddress.IPv6Address(pb_vnet_dir["overlay_ip"]).packed

    if json_obj.get("appliance") is not None:
        if pb.action_type != RoutingType.ROUTING_TYPE_APPLIANCE and pb.routing_type != RoutingType.ROUTING_TYPE_APPLIANCE:
            raise Exception("Action or routing type needs to be appliance for this attribute")
        pb.appliance = json_obj["appliance"]

    if json_obj.get("service_tunnel") is not None:
        if pb.action_type != RoutingType.ROUTING_TYPE_SERVICETUNNEL and pb.routing_type != RoutingType.ROUTING_TYPE_SERVICETUNNEL:
            raise Exception("Action or routing type needs to be service_tunnel for this attribute")

        pb_serv_tunnel = json_obj["service_tunnel"]
        if pb_serv_tunnel.get("overlay_sip") is not None:
            if type(IP(pb_serv_tunnel["overlay_sip"])) is ipaddress.IPv4Address:
                pb.service_tunnel.overlay_sip.ipv4 = socket.htonl(int(ipaddress.IPv4Address(pb_serv_tunnel["overlay_sip"])))
            elif type(IP(pb_serv_tunnel["overlay_sip"])) is ipaddress.IPv6Address:
                pb.service_tunnel.overlay_sip.ipv6 = ipaddress.IPv6Address(pb_serv_tunnel["overlay_sip"]).packed

        if pb_serv_tunnel.get("overlay_dip") is not None:
            if type(IP(pb_serv_tunnel["overlay_dip"])) is ipaddress.IPv4Address:
                pb.service_tunnel.overlay_dip.ipv4 = socket.htonl(int(ipaddress.IPv4Address(pb_serv_tunnel["overlay_dip"])))
            elif type(IP(pb_serv_tunnel["overlay_dip"])) is ipaddress.IPv6Address:
                pb.service_tunnel.overlay_dip.ipv6 = ipaddress.IPv6Address(pb_serv_tunnel["overlay_dip"]).packed

        if pb_serv_tunnel.get("underlay_sip") is not None:
            if type(IP(pb_serv_tunnel["underlay_sip"])) is ipaddress.IPv4Address:
                pb.service_tunnel.underlay_sip.ipv4 = socket.htonl(int(ipaddress.IPv4Address(pb_serv_tunnel["underlay_sip"])))
            elif type(IP(pb_serv_tunnel["underlay_sip"])) is ipaddress.IPv6Address:
                pb.service_tunnel.underlay_sip.ipv6 = ipaddress.IPv6Address(pb_serv_tunnel["underlay_sip"]).packed

        if pb_serv_tunnel.get("underlay_dip") is not None:
            if type(IP(pb_serv_tunnel["underlay_dip"])) is ipaddress.IPv4Address:
                pb.service_tunnel.underlay_dip.ipv4 = socket.htonl(int(ipaddress.IPv4Address(pb_serv_tunnel["underlay_dip"])))
            elif type(IP(pb_serv_tunnel["underlay_dip"])) is ipaddress.IPv6Address:
                pb.service_tunnel.underlay_dip.ipv6 = ipaddress.IPv6Address(pb_serv_tunnel["underlay_dip"]).packed

        if pb_serv_tunnel.get("overlay_sip_prefix") is not None:
            if type(IP(pb_serv_tunnel["overlay_sip_prefix"])) is ipaddress.IPv4Address:
                pb.service_tunnel.overlay_sip_prefix.ipv4 = socket.htonl(int(ipaddress.IPv4Address(pb_serv_tunnel["overlay_sip_prefix"])))
            elif type(IP(pb_serv_tunnel["overlay_sip_prefix"])) is ipaddress.IPv6Address:
                pb.service_tunnel.overlay_sip_prefix.ipv6 = ipaddress.IPv6Address(pb_serv_tunnel["overlay_sip_prefix"]).packed

            if type(IP(pb_serv_tunnel["overlay_dip_prefix"])) is ipaddress.IPv4Address:
                pb.service_tunnel.overlay_dip_prefix.ipv4 = socket.htonl(int(ipaddress.IPv4Address(pb_serv_tunnel["overlay_dip_prefix"])))
            elif type(IP(pb_serv_tunnel["overlay_dip_prefix"])) is ipaddress.IPv6Address:

    if json_obj.get("metering_policy_en") is not None:
        pb.metering_class = json_obj["metering_policy_en"] == "true"

'''
message RouteTypeItem {
    string action_name = 1;
    route_type.ActionType action_type = 2;
    // Optional
    // encap type depends on the action_type - {vxlan, nvgre}
    optional route_type.EncapType encap_type = 3;
    // Optional

    repeated RouteTypeItem items = 1;

def routing_type_from_json(json_obj):
    pb = RouteType()
    if isinstance(json_obj, list):
        for item in json_obj:
            pbi = RouteTypeItem()
            pbi.action_name = item["action_name"]
            pbi.action_type = get_action_type_from_str(item["action_type"])
            if item.get("encap_type") is not None:
                pbi.encap_type = get_encap_type_from_str(item.get("encap_type"))
            if item.get("vni") is not None:
                pbi.vni = int(item["vni"])
            pb.items.append(pbi)
    else:
        pbi = RouteTypeItem()
        pbi.action_name = json_obj["action_name"]
        pbi.action_type = get_action_type_from_str(json_obj["action_type"])
        if json_obj.get("encap_type") is not None:
            if json_obj["encap_type"] == "vxlan":
                pbi.encap_type = 1
            elif json_obj["encap_type"] == "nvgre":
                pbi.encap_type = 2
            else:
                pbi.encap_type = 0
        if json_obj.get("vni") is not None:
            pbi.vni = int(json_obj["vni"])
        pb.items.append(pbi)
    return pb

def acl_rule_from_json(json_obj):
    pb = AclRule()
    pb.priority = int(json_obj["priority"])
    if json_obj.get("action") == "allow":
        pb.action = Action.ACTION_PERMIT
    elif json_obj.get("action") == "deny":
        pb.action = Action.ACTION_DENY
    else:
        pb.action = Action.ACTION_UNSPECIFIED

    pb.terminating = json_obj["terminating"] == "true"

    if json_obj.get("protocol") is not None:
        protocols = [i.strip() for i in json_obj["protocol"].split(',')]
        for i in range(0, len(protocols)):
            protocols[i] = int(protocols[i])
        pb.protocol.extend(protocols)

    if json_obj.get("src_tag") is not None:
        src_tags = [i.strip() for i in json_obj["src_tag"].split(',')]
        pb.src_tag.extend(src_tags)

    if json_obj.get("dst_tag") is not None:
        dst_tags = [i.strip() for i in json_obj["dst_tag"].split(',')]
        pb.dst_tag.extend(dst_tags)

    if json_obj.get("src_addr") is not None:
        addr_list = [i.strip() for i in json_obj["src_addr"].split(',')]
        for i in range(0, len(addr_list)):
            src_addr_prefix = [i.strip() for i in addr_list[i].split("/")]
            src_ip_prefix = IpPrefix()
            src_ip_prefix.ip.ipv4 = socket.htonl(int(ipaddress.IPv4Address(src_addr_prefix[0])))
            if len(src_addr_prefix) == 1:
                src_ip_prefix.mask.ipv4 = 32
            else:
                src_ip_prefix.mask.ipv4 = int(src_addr_prefix[1])
            pb.src_addr.append(src_ip_prefix)

    if json_obj.get("dst_addr") is not None:
        addr_list = [i.strip() for i in json_obj["dst_addr"].split(',')]
        for i in range(0, len(addr_list)):
            dst_addr_prefix = [i.strip() for i in addr_list[i].split('/')]
            dst_ip_prefix = IpPrefix()

    repeated types.IpAddress endpoints = 1;
    route_type.EncapType encap_type = 2;

    return pb

def json_to_proto(key, json_obj):
    table_name = re.search(r"DASH_(\w+)_TABLE", key).group(1)
    if table_name in handlers_map:
        pb = handlers_map[table_name](json_obj)
    else:
        raise Exception("Unknown table %s" % table_name)
    return pb.SerializeToString()

def tbl_name_to_type(tbl_name):
    dash_name = re.search(r"DASH_(\w+)_TABLE", tbl_name).group(1)
    # Split the string by underscores
    words = dash_name.split('_')
    # Capitalize the first character of each word
    words = [word.capitalize() for word in words]
    return ''.join(words)

def from_pb(tbl_name, byte_array):
    type_name = tbl_name_to_type(tbl_name)
    obj_type = globals()[type_name]
    obj = obj_type()
    obj.ParseFromString(byte_array)
    return obj
